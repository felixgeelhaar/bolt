Zerolog (Before)

package main

import (
    "os"
    "github.com/rs/zerolog"
)

func main() {
    logger := zerolog.New(os.Stdout).With().Timestamp().Logger()
    
    logger.Info().
        Str("service", "api").
        Int("port", 8080).
        Msg("Server starting")
}

Bolt (After)

package main

import (
    "os"
    "github.com/felixgeelhaar/bolt"
)

func main() {
    logger := bolt.New(bolt.NewJSONHandler(os.Stdout))
    
    logger.Info().
        Str("service", "api").
        Int("port", 8080).
        Msg("Server starting")
}
